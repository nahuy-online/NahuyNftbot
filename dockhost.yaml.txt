# Здесь мы описываем, как хостинг должен видеть наше приложение

  domains:
    # Объявляем динамический домен (хостинг сам выдаст имя, например, nft-bot.dockhost.net)
    app-domain: ~ 

  routes:
    # Создаем маршрут для веб-трафика
    web-route:
      domain: app-domain # Используем объявленный выше домен
      path: /
      # ВАЖНО: Указываем имя сервиса как 'nft-bot' (как на вашем скриншоте)
      # Если деплой не сработает, попробуйте изменить это имя на '.' или имя проекта
      service: nft-bot
      port: 80
      type: http
